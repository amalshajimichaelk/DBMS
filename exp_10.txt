Create Or Replace Function Factorial(N Int)
Return Int Is
  Fact Int:=1;
Begin
  For I In 1..N Loop
    Fact := Fact*I;
  End Loop;
  Return Fact;
End;
/
Declare
  Result Int;
Begin
  Result := Factorial(5);
  Dbms_Output.Put_Line('Factorial is ' || Result);
End;
/
